# Elbow Plot for K-Means Clustering
def plot_elbow():
    features = data[['Age', 'Cholesterol', 'MaxHR']]
    wcss = []

    for k in range(1, 11):
        kmeans = KMeans(n_clusters=k, random_state=42)
        kmeans.fit(features)
        wcss.append(kmeans.inertia_)

    plt.figure(figsize=(8, 5))
    plt.plot(range(1, 11), wcss, marker='o', linestyle='--')
    plt.title('Elbow Method for Optimal K')
    plt.xlabel('Number of Clusters')
    plt.ylabel('WCSS')
    plt.show()

plot_elbow()

# K-Means Clustering
def perform_kmeans():
    features = data[['Age', 'Cholesterol', 'MaxHR']]
    kmeans = KMeans(n_clusters=3, random_state=42)
    data['Cluster'] = kmeans.fit_predict(features)

    plt.figure(figsize=(8, 5))
    sns.scatterplot(x='Cholesterol', y='MaxHR', hue='Cluster', palette='viridis', data=data)
    plt.title('K-Means Clustering (3 Clusters)')
    plt.xlabel('Cholesterol')
    plt.ylabel('MaxHR')
    plt.legend(title='Cluster')
    plt.show()

perform_kmeans()

# Line Fitting: Cholesterol vs Age
def perform_line_fitting():
    X = data['Age'].values.reshape(-1, 1)
    y = data['Cholesterol'].values

    model = LinearRegression()
    model.fit(X, y)

    y_pred = model.predict(X)

    plt.figure(figsize=(8, 5))
    plt.scatter(data['Age'], data['Cholesterol'], color='blue', label='Actual Data')
    plt.plot(data['Age'], y_pred, color='red', label='Fitted Line')
    plt.title('Line Fitting: Cholesterol vs Age')
    plt.xlabel('Age')
    plt.ylabel('Cholesterol')
    plt.legend()
    plt.show()

    mse = mean_squared_error(y, y_pred)
    print(f"Mean Squared Error: {mse}")

perform_line_fitting()
